        // --- DYNAMIC PAGE LOGIC ---
        document.addEventListener('DOMContentLoaded', () => {
            const bookGrid = document.getElementById('book-grid');
            const filtersContainer = document.getElementById('filters-container');
            const modal = document.getElementById('book-modal');
            const modalContent = document.getElementById('modal-content');
            let currentFilter = 'All';

            const categories = ['All', ...new Set(books.map(book => book.category))];

            function renderBooks() {
                bookGrid.innerHTML = '';
                const filteredBooks = books.filter(book => currentFilter === 'All' || book.category === currentFilter);
                
                filteredBooks.forEach((book, index) => {
                    const bookDataIndex = books.indexOf(book); // Get original index for modal
                    const card = `
                        <div class="space-y-2">
                            <button data-index="${bookDataIndex}" class="book-card w-full aspect-[2/3] bg-gray-200 dark:bg-gray-800 rounded-md overflow-hidden transition-transform duration-200 hover:scale-105">
                                <img src="${book.image}" alt="Cover of ${book.title}" class="w-full h-full object-cover">
                            </button>
                            <div class="text-center">
                                <p class="font-bold font-display text-sm truncate">${book.title}</p>
                                <p class="text-xs text-ink/70 dark:text-canvas/70">${book.author}</p>
                            </div>
                        </div>
                    `;
                    bookGrid.innerHTML += card;
                });

                // Add event listeners to the new book cards
                document.querySelectorAll('.book-card').forEach(card => {
                    card.addEventListener('click', openModal);
                });
            }

            function renderFilters() {
                filtersContainer.innerHTML = '';
                categories.forEach(category => {
                    const button = document.createElement('button');
                    button.textContent = category;
                    button.dataset.category = category;
                    button.className = `filter-btn font-ui font-medium py-2 px-4 rounded-full text-sm transition-colors ${currentFilter === category ? 'active' : 'bg-gray-200/80 dark:bg-gray-800/80 hover:bg-gray-300 dark:hover:bg-gray-700'}`;
                    
                    button.addEventListener('click', () => {
                        currentFilter = category;
                        renderFilters();
                        renderBooks();
                    });
                    
                    filtersContainer.appendChild(button);
                });
            }

            function openModal(event) {
                const index = event.currentTarget.dataset.index;
                const book = books[index];
                
                modalContent.innerHTML = `
                    <div class="w-full md:w-1/3 flex-shrink-0">
                        <img src="${book.image}" alt="Cover of ${book.title}" class="w-full h-full object-cover">
                    </div>
                    <div class="p-6 flex flex-col overflow-y-auto">
                        <div class="flex-grow">
                            <h2 id="modal-title" class="text-2xl font-bold font-display mb-1">${book.title}</h2>
                            <p class="text-md text-ink/70 dark:text-canvas/70 mb-4">${book.author}</p>
                            <p class="font-body leading-relaxed text-ink/90 dark:text-canvas/90">${book.description}</p>
                            <a href="#" class="inline-block mt-4 text-sm font-bold text-glow hover:underline">View Details (Google Books) &rarr;</a>
                        </div>
                        <button id="modal-close-button" class="mt-6 self-start md:self-end bg-gray-200 dark:bg-gray-700 text-ink dark:text-canvas font-bold font-ui py-2 px-4 rounded-lg text-sm hover:opacity-90 transition-opacity">Close</button>
                    </div>
                `;
                
                modal.classList.remove('hidden');
                document.body.style.overflow = 'hidden'; // Prevent background scrolling
                document.getElementById('modal-close-button').addEventListener('click', closeModal);
            }

            function closeModal() {
                modal.classList.add('hidden');
                document.body.style.overflow = 'auto';
            }

            modal.addEventListener('click', (event) => {
                if (event.target === modal) {
                    closeModal();
                }
            });

            document.addEventListener('keydown', (event) => {
                if (event.key === 'Escape' && !modal.classList.contains('hidden')) {
                    closeModal();
                }
            });

            renderFilters();
            renderBooks();
        });


        // --- STANDARD SITE-WIDE LOGIC ---
        
        // Mobile Menu Logic
        const mobileMenuButton = document.getElementById('mobile-menu-button');
        const mobileMenu = document.getElementById('mobile-menu');
        mobileMenuButton.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        // Dark Mode Logic
        const themeToggleDarkIcon = document.getElementById('theme-toggle-dark-icon');
        const themeToggleLightIcon = document.getElementById('theme-toggle-light-icon');
        const themeToggleButton = document.getElementById('theme-toggle');
        function setTheme(isDark) {
            if (isDark) {
                document.documentElement.classList.add('dark');
                themeToggleLightIcon.classList.remove('hidden');
                themeToggleDarkIcon.classList.add('hidden');
                localStorage.setItem('color-theme', 'dark');
            } else {
                document.documentElement.classList.remove('dark');
                themeToggleLightIcon.classList.add('hidden');
                themeToggleDarkIcon.classList.remove('hidden');
                localStorage.setItem('color-theme', 'light');
            }
        }
        if (localStorage.getItem('color-theme') === 'dark' || (!('color-theme' in localStorage) && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
            setTheme(true);
        } else {
            setTheme(false);
        }
        themeToggleButton.addEventListener('click', () => {
            setTheme(!document.documentElement.classList.contains('dark'));
        });
        
        // Search Bar Logic
        const searchToggleButton = document.getElementById('search-toggle-button');
        const searchInput = document.getElementById('search-input');
        searchToggleButton.addEventListener('click', (event) => {
            searchInput.classList.toggle('hidden');
            if (!searchInput.classList.contains('hidden')) {
                searchInput.focus();
            }
            event.stopPropagation();
        });
        document.addEventListener('click', (event) => {
            const searchContainer = document.getElementById('search-container');
            if (searchContainer && !searchContainer.contains(event.target) && !searchInput.classList.contains('hidden')) {
                searchInput.classList.add('hidden');
            }
        });

        // Dynamic Email Logic
        const emailUser = 'message';
        const emailDomain = 'aiwritersretreat.com';
        const emailElement = document.getElementById('contact-email');
        if(emailElement) {
            emailElement.setAttribute('href', `mailto:${emailUser}@${emailDomain}`);
        }
        
        // "More" Menu Dropdown Logic
        const moreMenuContainer = document.getElementById('more-menu-container');
        const moreMenuButton = document.getElementById('more-menu-button');
        const moreMenuDropdown = document.getElementById('more-menu-dropdown');

        if (moreMenuButton) {
            moreMenuButton.addEventListener('click', (event) => {
                moreMenuDropdown.classList.toggle('hidden');
                event.stopPropagation();
            });
        }

        document.addEventListener('click', (event) => {
            if (moreMenuContainer && !moreMenuContainer.contains(event.target) && !moreMenuDropdown.classList.contains('hidden')) {
                moreMenuDropdown.classList.add('hidden');
            }
        });

    </script>
